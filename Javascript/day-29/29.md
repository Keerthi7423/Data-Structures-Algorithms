----------------------------------------
🟢 BASIC
----------------------------------------

1) let vs const
----------------------------------------
let a = 10;
a = 20;        // allowed

const b = 10;
// b = 20;     // ❌ error

Explanation:
- let → can reassign
- const → cannot reassign

----------------------------------------

2) Block scope
----------------------------------------
if (true) {
    let x = 10;
}
// console.log(x); ❌ error

Explanation:
- let/const are block scoped

----------------------------------------

3) Arrow function
----------------------------------------
let add = (a, b) => a + b;

console.log(add(2, 3));

Answer:
5

----------------------------------------

4) Template literals
----------------------------------------
let name = "Keerthi";

console.log(`Hello ${name}`);

Answer:
Hello Keerthi

----------------------------------------

----------------------------------------
🟡 INTERMEDIATE
----------------------------------------

5) Default parameters
----------------------------------------
function greet(name = "Guest") {
    return `Hello ${name}`;
}

console.log(greet());

Answer:
Hello Guest

----------------------------------------

6) Destructuring (object)
----------------------------------------
let user = {name:"Keerthi", age:22};

let {name, age} = user;

console.log(name);

Answer:
Keerthi

----------------------------------------

7) Destructuring (array)
----------------------------------------
let arr = [10, 20, 30];

let [a, b] = arr;

console.log(b);

Answer:
20

----------------------------------------

8) Spread operator (array)
----------------------------------------
let arr1 = [1,2];
let arr2 = [...arr1, 3, 4];

console.log(arr2);

Answer:
[1,2,3,4]

----------------------------------------

9) Spread (object)
----------------------------------------
let obj1 = {a:1};
let obj2 = {...obj1, b:2};

console.log(obj2);

Answer:
{a:1, b:2}

----------------------------------------

10) Rest parameter
----------------------------------------
function sum(...nums) {
    return nums.reduce((a,b) => a+b, 0);
}

console.log(sum(1,2,3));

Answer:
6

----------------------------------------

----------------------------------------
🔴 LOGIC BUILDING
----------------------------------------

11) Swap variables
----------------------------------------
let a = 5, b = 10;

[a, b] = [b, a];

console.log(a, b);

Answer:
10 5

----------------------------------------

12) Clone array
----------------------------------------
let arr = [1,2,3];
let copy = [...arr];

Explanation:
- Spread creates shallow copy

----------------------------------------

13) Merge arrays
----------------------------------------
let a = [1,2];
let b = [3,4];

let merged = [...a, ...b];

----------------------------------------

14) Extract specific values
----------------------------------------
let user = {
    name:"Keerthi",
    age:22,
    city:"Bangalore"
};

let {name, city} = user;

----------------------------------------

15) Function with destructuring
----------------------------------------
function printUser({name, age}) {
    console.log(name, age);
}

printUser({name:"Keerthi", age:22});

----------------------------------------

----------------------------------------
🧠 ADVANCED (INTERVIEW 🔥)
----------------------------------------

16) Nested destructuring
----------------------------------------
let obj = {
    user: {
        name: "Keerthi"
    }
};

let {user: {name}} = obj;

console.log(name);

----------------------------------------

17) Rest in destructuring
----------------------------------------
let obj = {a:1, b:2, c:3};

let {a, ...rest} = obj;

console.log(rest);

Answer:
{b:2, c:3}

----------------------------------------

18) Template with expression
----------------------------------------
let a = 5, b = 10;

console.log(`Sum = ${a + b}`);

Answer:
Sum = 15

----------------------------------------

19) Arrow + map
----------------------------------------
let arr = [1,2,3];

let result = arr.map(x => x * 2);

console.log(result);

Answer:
[2,4,6]

----------------------------------------

20) Arrow + filter
----------------------------------------
let arr = [1,2,3,4];

let even = arr.filter(x => x % 2 === 0);

console.log(even);

Answer:
[2,4]

----------------------------------------

----------------------------------------
KEY SUMMARY
----------------------------------------

let → reassignable
const → fixed reference

Arrow → short functions
Template → `${}`

Destructuring → extract values
Spread → expand / copy
Rest → collect values

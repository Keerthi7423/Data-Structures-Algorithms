OBJECT + JSON PRACTICE

----------------------------------------
🟢 ROUND 1: BASIC
----------------------------------------

1) Object → JSON
----------------------------------------
let obj = {name:"Keerthi", age:22};

let json = JSON.stringify(obj);
console.log(json);

Answer:
'{"name":"Keerthi","age":22}'

Explanation:
- Converts object into JSON string

----------------------------------------

2) JSON → Object
----------------------------------------
let json = '{"name":"Keerthi","age":22}';

let obj = JSON.parse(json);
console.log(obj);

Answer:
{ name: "Keerthi", age: 22 }

Explanation:
- Converts string into usable object

----------------------------------------

3) Access property after parsing
----------------------------------------
let json = '{"name":"Keerthi"}';

let obj = JSON.parse(json);

console.log(obj.name);

Answer:
Keerthi

----------------------------------------

4) Add new property
----------------------------------------
obj.city = "Bangalore";
console.log(obj);

Answer:
{ name:"Keerthi", city:"Bangalore" }

----------------------------------------

5) Delete property
----------------------------------------
delete obj.name;
console.log(obj);

Answer:
{ city:"Bangalore" }

----------------------------------------

🟡 ROUND 2: LOOP + JSON
----------------------------------------

6) Print keys
----------------------------------------
let json = '{"a":1,"b":2}';
let obj = JSON.parse(json);

for(let key in obj){
    console.log(key);
}

Answer:
a
b

----------------------------------------

7) Print values
----------------------------------------
for(let key in obj){
    console.log(obj[key]);
}

Answer:
1
2

----------------------------------------

8) Print key-value pairs
----------------------------------------
for(let key in obj){
    console.log(key + ": " + obj[key]);
}

Answer:
a: 1
b: 2

----------------------------------------

9) Count properties
----------------------------------------
let count = Object.keys(obj).length;
console.log(count);

Answer:
2

----------------------------------------

10) Check key exists
----------------------------------------
console.log(obj.hasOwnProperty("a"));

Answer:
true

🔴 ROUND 3: LOGIC BUILDING
----------------------------------------

11) Sum numeric values
----------------------------------------
let sum = 0;

for(let key in obj){
    sum += obj[key];
}

console.log(sum);

Answer:
(sum of values)

----------------------------------------

12) Find max value
----------------------------------------
let max = -Infinity;

for(let key in obj){
    if(obj[key] > max){
        max = obj[key];
    }
}

console.log(max);

----------------------------------------

13) Print only string values
----------------------------------------
let json = '{"a":"hi","b":10,"c":"js"}';
let obj = JSON.parse(json);

for(let key in obj){
    if(typeof obj[key] === "string"){
        console.log(obj[key]);
    }
}

Answer:
hi
js

----------------------------------------

14) Convert strings to uppercase
----------------------------------------
for(let key in obj){
    if(typeof obj[key] === "string"){
        obj[key] = obj[key].toUpperCase();
    }
}

console.log(obj);

----------------------------------------

15) New object with only numbers
----------------------------------------
let result = {};

for(let key in obj){
    if(typeof obj[key] === "number"){
        result[key] = obj[key];
    }
}

console.log(result);

🧠 ROUND 4: ARRAY + JSON
----------------------------------------

16) Array of objects → JSON
----------------------------------------
let arr = [
    {name:"A"},
    {name:"B"}
];

let json = JSON.stringify(arr);
console.log(json);

----------------------------------------

17) Parse and print names
----------------------------------------
let json = '[{"name":"A"},{"name":"B"}]';

let arr = JSON.parse(json);

for(let i = 0; i < arr.length; i++){
    console.log(arr[i].name);
}

Answer:
A
B

----------------------------------------

18) Find highest score
----------------------------------------
let json = '[{"name":"A","score":50},{"name":"B","score":80}]';

let arr = JSON.parse(json);

let max = 0;

for(let i = 0; i < arr.length; i++){
    if(arr[i].score > max){
        max = arr[i].score;
    }
}

console.log(max);

Answer:
80

----------------------------------------

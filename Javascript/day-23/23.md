----------------------------------------
1) Create object with method
----------------------------------------
Question:
Create an object with name and a method to print it

Code:
let person = {
    name: "Keerthi",
    showName: function() {
        console.log(this.name);
    }
};

person.showName();

Answer:
Keerthi

Explanation:
- this.name refers to object's name property

----------------------------------------
2) Add method to existing object
----------------------------------------
Question:
Add a method to print greeting

Code:
let obj = {
    name: "JS"
};

obj.greet = function() {
    console.log("Hello " + this.name);
};

obj.greet();

Answer:
Hello JS

3) Object.keys()
----------------------------------------
Question:
Get all keys

Code:
let obj = {a:1, b:2, c:3};
console.log(Object.keys(obj));

Answer:
["a", "b", "c"]

----------------------------------------
4) Object.values()
----------------------------------------
Question:
Get all values

Code:
let obj = {a:1, b:2};
console.log(Object.values(obj));

Answer:
[1, 2]

----------------------------------------
5) Object.entries()
----------------------------------------
Question:
Get key-value pairs

Code:
let obj = {a:1, b:2};
console.log(Object.entries(obj));

Answer:
[["a",1], ["b",2]]

----------------------------------------
6) Loop using Object.keys()
----------------------------------------
Question:
Print all keys and values

Code:
let obj = {name:"Keerthi", age:22};

Object.keys(obj).forEach(key => {
    console.log(key, obj[key]);
});

Answer:
name Keerthi
age 22

8) Merge objects
----------------------------------------
Question:
Merge two objects

Code:
let obj1 = {a:1};
let obj2 = {b:2};

let result = Object.assign({}, obj1, obj2);
console.log(result);

Answer:
{a:1, b:2}

----------------------------------------
9) Freeze object
----------------------------------------
Question:
Prevent changes

Code:
let obj = {a:1};
Object.freeze(obj);

obj.a = 10;
console.log(obj);

Answer:
{a:1}

Explanation:
- Value does not change

----------------------------------------
10) Seal object
----------------------------------------
Question:
Allow update but not add/remove

Code:
let obj = {a:1};
Object.seal(obj);

obj.a = 5;
obj.b = 10;

console.log(obj);

Answer:
{a:5}

Explanation:
- a updated
- b not added

13) Sum of values
----------------------------------------
Question:
Find sum of all values

Code:
let obj = {a:10, b:20, c:30};

let sum = 0;

Object.values(obj).forEach(val => {
    sum += val;
});

console.log(sum);

Answer:
60

----------------------------------------
14) Convert values to uppercase
----------------------------------------
Question:
Uppercase all string values

Code:
let obj = {a:"hi", b:"hello"};

Object.keys(obj).forEach(key => {
    obj[key] = obj[key].toUpperCase();
});

console.log(obj);

Answer:
{a:"HI", b:"HELLO"}

----------------------------------------
15) Find max value
----------------------------------------
Question:
Find largest value

Code:
let obj = {a:10, b:50, c:30};

let max = -Infinity;

Object.values(obj).forEach(val => {
    if(val > max){
        max = val;
    }
});

console.log(max);

Answer:
50

----------------------------------------
FINAL SUMMARY
----------------------------------------

Object methods = functions inside objects
this → current object
Object.keys() → keys
Object.values() → values
Object.entries() → pairs
assign() → merge
freeze() → lock object
seal() → partial lock
